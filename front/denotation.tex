\begin{denotation}[4cm]

\item[AMM]{
  Automated Market Maker.
}

\item[API]{
  Application Programming Interface.
}

\item[CFG]{
  Control-Flow Graph.
}

\item[CSV]{
  Comma-Separated Values.
}

\item[DeFi]{
  Decentralized Finance.
}

\item[EVM]{
  Ethereum Virtual Machine.
}

\item[F1-score]{
  The harmonic mean of precision and recall.
}

\item[IR]{
  Intermediate Representation.
}

\item[JSON]{
  JavaScript Object Notation.
}

\item[LLM]{
  Large Language Model.
}

\item[PMDetector]{
  A prior hybrid framework for detecting price manipulation vulnerabilities in DeFi protocols.
}

\item[RQ]{
  Research Question.
}

\item[SSA]{
  Static Single Assignment.
}

\item[Slither]{
  A static analysis framework for Solidity smart contracts.
}

\item[SlithIR]{
  The intermediate representation exposed by Slither.
}

\item[TWAP]{
  Time-Weighted Average Price.
}

\item[$p$]{
  A project in the over-reporting penalty, or a predicted explanation in the semantic explanation score, depending on context.
}

\item[$g$]{
  A ground-truth audit-finding explanation in the semantic explanation score.
}

\item[$E(\cdot)$]{
  The embedding function used to map report text into a vector representation.
}

\item[$\tau$]{
  The semantic similarity threshold used when computing the description score.
}

\item[$\cos(E(p), E(g))$]{
  Cosine similarity between the predicted explanation embedding and the ground-truth explanation embedding.
}

\item[$\mathrm{DescriptionScore}(p,g)$]{
  The semantic explanation score between a predicted report and a ground-truth finding.
}

\item[$\mathrm{Penalty}(p)$]{
  The over-reporting penalty for project $p$.
}

\item[$\#\mathrm{Predictions}(p)$]{
  The number of generated findings for project $p$.
}

\item[$\#\mathrm{GroundTruth}(p)$]{
  The number of relevant ground-truth audit findings for project $p$.
}

\end{denotation}
